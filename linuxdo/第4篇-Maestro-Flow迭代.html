安装

npm install -g maestro-flow #需要使用Maestro delegate 多cli功能

npm install -g maestro-flow-one 

maestro-flow install #默认安装到全局

codex配置

[features]
enable_fanout = true
multi_agent_v2=true
default_mode_request_user_input = true
goals = true
